// splice특정 항목을 제거할 때 사용함. 해당 값을 변수에 저장 한다. const numbers = [10, 20, 30, 40, 50, 60, 70, 80]; const index = numbers.indexOf(30); console.log(index); numbers.splice(index, 1); //indexOf가 설정된 숫자부터 제거 indexOf의 2부터 1개 지우겠다, 2개면 2개 지우겠다. console.log(numbers); const numbers1 = [10, 20, 30, 40, 50, 60, 70, 80]; const index1 = numbers1.indexOf(30); const spliced = numbers1.splice(index1, 2); console.log(index1); console.log(spliced); console.log(numbers1); // 배열을 잘라날떄 slice 사용 const numbers2 = [10, 20, 30, 40, 50, 60, 70, 80]; const sliced = numbers2.slice(0, 4); // 시작 index ~ 엔드 index 까지 잘라내는 범위를 숫자로 입력 console.log(sliced); console.log(numbers2); // splice는 기존의 배열의 값을 아예 삭제하는데 slice는 기존 배열의 값에 영향을 끼치지 않는다.
